1  `include "data_defs.v"
     2  `include "Packet.sv"
     3  class Generator;
     4  	string  name;		
     5  	Packet  pkt2send;	
     6  
     7     	typedef mailbox #(Packet) in_box_type;
     8  	in_box_type in_box;
     9  	
     10  	int		packet_number;
     11  	int 	number_packets;
     12  	extern function new(string name = "Generator", int number_packets);
     13  	extern virtual task gen();
     14  	extern virtual task start();
     15  endclass
     16  
     17  function Generator::new(string name = "Generator", int number_packets);
     18  	this.name = name;
     19  	this.pkt2send = new();
     20  	this.in_box = new;
     21  	this.packet_number = 0;
     22  	this.number_packets = number_packets;
     23  endfunction
     24  
     25  task Generator::gen();
     26  	pkt2send.name = $psprintf("Packet[%0d]", packet_number++);
     27  	if (!pkt2send.randomize()) 
     28  	begin
     29  		$display(" \n%m\n[ERROR]%0d gen(): Randomization Failed!", $time);
     30  		$finish;	
     31  	end
     32  	pkt2send.enable = $urandom_range(0,1);
     33  	$display ($time, " [GENERATOR] Packet Generation done .. Now to put it in Driver mailbox");	
     34  endtask
     35  
     36  task Generator::start();
     37  	  $display ($time, "ns:  [GENERATOR] Generator Started");
     38  	  fork
     39  		  for (int i=0; i<number_packets || number_packets <= 0; i++) 
     40  		  begin
     41  			  gen();
     42  			  begin 
     43  			      Packet pkt = new pkt2send; 
     44  			      in_box.put(pkt); // FUNNY .. 
     45  			  end
     46  		  end
     47  		  $display($time, "ns:  [GENERATOR] Generation Finished Creating %d Packets  ", number_packets);
     48        join_none
     49  endtask
