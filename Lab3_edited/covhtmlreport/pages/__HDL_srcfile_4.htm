     2  module Ex_Preproc  (   clock, reset, enable_ex, src1, src2, imm, control_in, 
     3                      mem_data_read_in, mem_data_write_out, mem_write_en,

     6                  );
     7  
     8      parameter   instr_wd  	=   `INSTR_WIDTH;
     9      parameter   reg_wd    	=   `REGISTER_WIDTH;

     11  
     12      input 					            clock, reset, enable_ex;
     13  	input  	 		[reg_wd -1:0]       src1, src2; 
     14  	input 	 		[reg_wd -1:0]       imm;
     15  	input 		    [6:0]			    control_in;
     16      // the data that was read in the Decode stage
     17  	input	 		[reg_wd	-1:0]		mem_data_read_in; 
     18      // data that is going to be written to the memory 
     19  	output  	    [reg_wd -1:0]       mem_data_write_out; 
     20      // enable for writing to memory   
     21  	output  	 	[reg_wd -1:0]  		aluin1, aluin2; 
     22  	output	        [2:0]			    operation_out;
     23  	output	        [2:0]               opselect_out;	
     24  	output          [4:0]          		shift_number;
     25      output                            	enable_shift, enable_arith;
     26  	
     27      output					            mem_write_en;  
     28  
     29  	// Intermediate Signals
     30  	reg					                mem_write_en;
     31  	
     32  	reg  	 		[reg_wd -1:0]  		aluin1, aluin2; 
     33  	reg 	        [2:0]			    opselect_out;
     34  	reg 	        [2:0]               operation_out;	
     35  	
     36  	wire	        [2:0]			    opselect;
     37  	wire				                immp_regn_op;
     38  	wire 	        [2:0]               operation;	
     39  
     40      reg          	[4:0]            	shift_number;
     41  	
     42      reg                                 enable_shift, enable_arith;
     43  	
     44  	assign	{operation, immp_regn_op,opselect} = control_in;
     45      // control[2:0] = opselect: selects type of operand to be used i.e. memory, register or immediate
     46      // control[3]  = immp_regn_op: 1=> immediate operation & 0: register operation
     47      // control[6:4] =  operation: selects type of operation to be performed i.e. arithmetic, shifting etc
